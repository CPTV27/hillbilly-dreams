### Step 4: Implement `ContentBlock.tsx`

This reusable component will be used for the "Listen," "Shows," and "Sponsor" sections. It will display a title, description, and an optional call-to-action button.

```typescript jsx
// apps/web/components/ContentBlock.tsx
import Link from 'next/link';
import React from 'react';

interface ContentBlockProps {
  title: string;
  description: string;
  ctaText?: string; // Optional CTA text
  ctaLink?: string; // Optional CTA link
  icon?: React.ReactNode; // Optional icon, e.g., an SVG or a React component
  reverseLayout?: boolean; // Optional prop to reverse text/icon order
}

export default function ContentBlock({
  title,
  description,
  ctaText,
  ctaLink,
  icon,
  reverseLayout = false,
}: ContentBlockProps) {
  const layoutClasses = reverseLayout ? 'md:flex-row-reverse' : 'md:flex-row';

  return (
    <section className="py-16 px-4 bg-gray-50 border-b border-gray-200 last:border-b-0">
      <div className={`max-w-5xl mx-auto flex flex-col ${layoutClasses} items-center gap-12`}>
        {icon && (
          <div className="flex-shrink-0 text-blue-600 w-24 h-24 md:w-32 md:h-32 flex items-center justify-center p-4 rounded-full bg-blue-100">
            {icon}
          </div>
        )}
        <div className="flex-grow text-center md:text-left">
          <h2 className="text-4xl font-bold text-gray-900 mb-4">
            {title}
          </h2>
          <p className="text-lg text-gray-700 mb-6 leading-relaxed">
            {description}
          </p>
          {ctaText && ctaLink && (
            <Link href={ctaLink} className="inline-block bg-blue-600 text-white hover:bg-blue-700 font-medium py-2 px-6 rounded-full transition duration-300 ease-in-out">
              {ctaText}
            </Link>
          )}
        </div>
      </div>
    </section>
  );
}
```
